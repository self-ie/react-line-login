## Install

```bash
npm install --save react-line-login
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-line-login'
import 'react-line-login/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
